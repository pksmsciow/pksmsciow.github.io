$b = 7;
$reszta = $a % $b;
echo "Reszta z dzielenia $a przez $b wynosi: <b>$reszta</b>";

echo "<hr><h2>Zadanie 2: Obliczenie wyrażenia</h2>";

$roznica_bitowa = $x_bit ^ $y_bit; // XOR

echo "Liczby: $x_bit oraz $y_bit<br>";
echo "Iloczyn bitowy (&): <b>$iloczyn_bitowy</b><br>";
echo "Suma bitowa (|): <b>$suma_bitowa</b><br>";
echo "Różnica bitowa (^): <b>$roznica_bitowa</b>";
