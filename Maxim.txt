<?php
// Definiowanie zmiennych (z przykładu 47.11)
$x = 12;
$y = 8;

// Obliczenia arytmetyczne
$suma = $x + $y;
$roznica = $x - $y;
$iloczyn = $x * $y;
$iloraz = $x / $y;
$reszta = $x % $y; // Operacja modulo

// Wyświetlanie wyników w przeglądarce
// Używamy <br>, aby każda informacja była w nowej linii
echo "Suma $x i $y wynosi $suma<br>";
echo "Różnica $x i $y wynosi $roznica<br>";
echo "Iloczyn $x i $y wynosi $iloczyn<br>";
echo "Iloraz $x przez $y wynosi $iloraz<br>";
echo "Reszta z dzielenia $x przez $y wynosi $reszta<br>";

echo "<hr>"; // Linia oddzielająca

// Przykład użycia operatorów porównania (z tabeli 47.3)
echo "Czy $x jest równe $y? ";
var_dump($x == $y); // Zwróci bool(false)
echo "<br>";

echo "Czy $x jest większe od $y? ";
var_dump($x > $y); // Zwróci bool(true)
?>
